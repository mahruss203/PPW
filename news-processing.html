import pandas as pd
import nltk
from nltk.corpus import stopwords

nltk.download("stopwords")

df = pd.read_csv("kompas_news.csv")

stop_words = set(stopwords.words("indonesian"))

def remove_stopwords(text):
    if pd.isnull(text):
        return ""
    tokens = text.split()
    filtered = [word for word in tokens if word.lower() not in stop_words]
    return " ".join(filtered)

df["Isi_Berita"] = df["Isi_Berita"].apply(remove_stopwords)

df.to_csv("news_stopword.csv", index=False)

import pandas as pd
import difflib

# Baca data
df = pd.read_csv("news_cleaned.csv")

# Load kamus kata
with open("indonesian-words.txt", "r", encoding="utf-8") as f:
    kamus = [line.strip() for line in f]

kamus_set = set(kamus)  # biar lookup cepat

# Fungsi koreksi dengan difflib
def correct_word(word):
    # Kalau kata ada di kamus → biarin
    if word in kamus_set:
        return word
    # Cari kata paling mirip
    match = difflib.get_close_matches(word, kamus, n=1, cutoff=0.8)
    return match[0] if match else word

# Cache hasil biar ga cek ulang
cache = {}
def correct_with_cache(word):
    if word not in cache:
        cache[word] = correct_word(word)
    return cache[word]

# Terapin ke dataset
df["Isi_Berita"] = df["Isi_Berita"].apply(
    lambda x: " ".join(correct_with_cache(w) for w in str(x).split())
)
