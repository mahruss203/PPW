import pandas as pd
import numpy as np
from gensim import corpora
from gensim.models import LdaModel

# === 1. Baca data mentah ===
df = pd.read_csv("news_tokenizing.csv")

if "berita" not in df.columns or "kategori" not in df.columns:
    raise ValueError("Kolom 'Isi Berita' atau 'Kategori' tidak ditemukan!")

# === 2. Tokenisasi ===
texts = df["berita"].astype(str).apply(lambda x: x.split()).tolist()

# === 3. Buat kamus & corpus ===
dictionary = corpora.Dictionary(texts)
dictionary.filter_extremes(no_below=2, no_above=0.8)
corpus = [dictionary.doc2bow(text) for text in texts]

# === 4. Latih model LDA ===
num_topics = 35
lda_model = LdaModel(
    corpus=corpus,
    id2word=dictionary,
    num_topics=num_topics,
    random_state=42,
    passes=10,
    iterations=100,
)

# === 5. Konversi ke vektor topik ===
def get_topic_vector(lda_model, bow, num_topics):
    topic_probs = [0.0] * num_topics
    for topic_id, prob in lda_model.get_document_topics(bow):
        topic_probs[topic_id] = prob
    return np.array(topic_probs)

X_lda = np.array([get_topic_vector(lda_model, bow, num_topics) for bow in corpus])

# === 6. Simpan hasil ke CSV ===
df_lda = pd.DataFrame(X_lda, columns=[f"Topic_{i+1}" for i in range(num_topics)])
df_lda["kategori"] = df["kategori"].values
df_lda.to_csv("news_lda_35.csv", index=False)

    top_words = [feature_names[j] for j in top_indices]
    print("Top words:", ", ".join(top_words))

# === 4. Optional: visualize clusters using PCA ===
pca = PCA(n_components=2)
reduced = pca.fit_transform(X)
plt.figure(figsize=(8,6))
plt.scatter(reduced[:,0], reduced[:,1], c=df_tfidf["Cluster"], cmap="coolwarm")
plt.title("K-Means Clustering of Spam vs Non-Spam (TF-IDF)")
plt.xlabel("PCA 1")
plt.ylabel("PCA 2")
plt.show()

# === 5. Save the clustered file ===
df_tfidf.to_csv("spam_kmeans.csv", index=False)
print("\n File 'spam_kmeans.csv' saved with cluster labels.")
