import pandas as pd
import nltk
from nltk.corpus import stopwords

nltk.download("stopwords")

df = pd.read_csv("pta_fix.csv")

stop_words = set(stopwords.words("indonesian"))

def remove_stopwords(text):
    if pd.isnull(text):
        return ""
    tokens = text.split()
    filtered = [word for word in tokens if word.lower() not in stop_words]
    return " ".join(filtered)

df["abstrak_id"] = df["abstrak_id"].apply(remove_stopwords)

df.to_csv("pta_stopword.csv", index=False)

    if pd.isna(text):

    text = str(text).lower()
