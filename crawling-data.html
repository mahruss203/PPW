import requests
					import pandas as pd
					
					# API key SpringerNature
					api_key = "123dac3851ae6344568d3195735a2dd5"
					
					# Daftar kata kunci
					keywords = ["web mining", "web usage mining"]
					
					# Endpoint API
					url = "https://api.springernature.com/meta/v2/json"
					
					# List untuk menampung semua hasil
					all_records = []
					
					# Loop tiap keyword
					for keyword in keywords:
					    params = {
					        "q": keyword,
					        "api_key": api_key,
					        "p": 10  # jumlah artikel per keyword
					    }
					
					    response = requests.get(url, params=params)
					
					    if response.status_code == 200:
					        data = response.json()
					        total = data['result'][0]['total']
					        print(f"\nðŸ”Ž Kata kunci '{keyword}' - total hasil: {total}\n")
					
					        for record in data['records']:
					            doi = record.get('doi', 'N/A')
					            title = record.get('title', 'No title')
					            abstract = record.get('abstract', 'No abstract')
					            publication = record.get('publicationName', 'N/A')
					            url_link = record.get('url', [{'value':'N/A'}])[0]['value']
					
					            # Simpan ke list
					            all_records.append({
					                "keyword": keyword,
					                "doi": doi,
					                "title": title,
					                "abstract": abstract,
					                "publication": publication,
					                "url": url_link
					            })
					
					            # Print ke terminal
					            print(f"DOI        : {doi}")
					            print(f"Title      : {title}")
					            print(f"Abstract   : {abstract}")
					            print(f"Publication: {publication}")
					            print(f"URL        : {url_link}\n")
					    else:
					        print("Error:", response.status_code, response.text)
					
					# Simpan ke CSV
					df = pd.DataFrame(all_records)
					df.to_csv("springer_crawling.csv", index=False, encoding="utf-8")
