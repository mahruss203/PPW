3. lower case dan Penghapusan Simbol# 

import pandas as pd
import re

# Load CSV
df = pd.read_csv("news_stopword.csv")

def clean_text(text):
    if pd.isna(text):
        return ""
    
    # Lowercase
    text = str(text).lower()
    
    # Hapus karakter non-alfabet kecuali titik dan spasi
    text = re.sub(r"[^a-z.\s]", " ", text)
    
    # Hapus spasi berlebih
    text = re.sub(r"\s+", " ", text).strip()
    
    # Kalau mau hapus stopword tertentu (opsional)
    words = text.split()
    cleaned_words = [w for w in words if w not in ["kompas.com", "cnnindonesia.com"]]  # contoh
    
    return " ".join(cleaned_words)

# Terapkan cleaning
df["Isi Berita"] = df["Isi Berita"].apply(clean_text)

# Buang baris kosong setelah cleaning
df = df[df["Isi Berita"] != ""]

# Simpan ke CSV baru
df.to_csv("news_cleaned.csv", index=False)

import pandas as pd
import difflib

# Baca data
df = pd.read_csv("news_cleaned.csv")

# Load kamus kata
with open("indonesian-words.txt", "r", encoding="utf-8") as f:
    kamus = [line.strip() for line in f]

kamus_set = set(kamus)  # biar lookup cepat

# Fungsi koreksi dengan difflib
def correct_word(word):
    # Kalau kata ada di kamus → biarin
    if word in kamus_set:
        return word
    # Cari kata paling mirip
    match = difflib.get_close_matches(word, kamus, n=1, cutoff=0.8)
    return match[0] if match else word

# Cache hasil biar ga cek ulang
cache = {}
def correct_with_cache(word):
    if word not in cache:
        cache[word] = correct_word(word)
    return cache[word]

# Terapin ke dataset
df["Isi Berita"] = df["Isi Berita"].apply(
    lambda x: " ".join(correct_with_cache(w) for w in str(x).split())
)

# Simpan hasil
df.to_csv("news_spellchecked.csv", index=False)
print("Selesai! Hasil ada di pta_spellchecked.csv")

7.2. CBOW # 

import pandas as pd
from gensim.models import Word2Vec
import numpy as np

# === 1. Baca data ===
df = pd.read_csv("news_tokenizing.csv")

# Tokenisasi sederhana (asumsi kolom sudah tokenized per kata dipisah spasi)
sentences = df["Isi Berita"].astype(str).apply(lambda x: x.split()).tolist()

# === 2. Latih model CBOW ===
model = Word2Vec(
    sentences,
    vector_size=100,   # dimensi embedding
    window=5,          # konteks sekitar kata
    min_count=2,       # kata muncul <2 kali di-skip
    sg=0,              # 0 = CBOW, 1 = SkipGram
    workers=4
)

# === 3. Representasi dokumen (rata-rata embedding kata) ===
def document_vector(words, model):
    # hanya ambil kata yg ada di vocab
    words = [w for w in words if w in model.wv.key_to_index]
    if len(words) == 0:
        return np.zeros(model.vector_size)
    return np.mean(model.wv[words], axis=0)

X_cbow = np.array([document_vector(doc, model) for doc in sentences])

# === 4. Buat DataFrame + tambahkan kategori ===
df_cbow = pd.DataFrame(X_cbow)

if "Kategori" in df.columns:
    df_cbow["Kategori"] = df["Kategori"].values

# Simpan hasil
df_cbow.to_csv("news_cbow.csv", index=False)

print("Hasil CBOW disimpan ke 'news_cbow.csv'")

7.3. LDA # 

import pandas as pd
import numpy as np
from gensim import corpora
from gensim.models import LdaModel

# === 1. Baca data mentah ===
df = pd.read_csv("news_tokenizing.csv")

if "Isi Berita" not in df.columns or "Kategori" not in df.columns:
    raise ValueError("Kolom 'Isi Berita' atau 'Kategori' tidak ditemukan!")

# === 2. Tokenisasi ===
texts = df["Isi Berita"].astype(str).apply(lambda x: x.split()).tolist()

# === 3. Buat kamus & corpus ===
dictionary = corpora.Dictionary(texts)
dictionary.filter_extremes(no_below=2, no_above=0.8)
corpus = [dictionary.doc2bow(text) for text in texts]

# === 4. Latih model LDA ===
num_topics = 20
lda_model = LdaModel(
    corpus=corpus,
    id2word=dictionary,
    num_topics=num_topics,
    random_state=42,
    passes=10,
    iterations=100,
)

# === 5. Konversi ke vektor topik ===
def get_topic_vector(lda_model, bow, num_topics):
    topic_probs = [0.0] * num_topics
    for topic_id, prob in lda_model.get_document_topics(bow):
        topic_probs[topic_id] = prob
    return np.array(topic_probs)

X_lda = np.array([get_topic_vector(lda_model, bow, num_topics) for bow in corpus])

# === 6. Simpan hasil ke CSV ===
df_lda = pd.DataFrame(X_lda, columns=[f"Topic_{i+1}" for i in range(num_topics)])
df_lda["Kategori"] = df["Kategori"].values
df_lda.to_csv("news_lda.csv", index=False)

print("✅ Hasil vektor LDA disimpan ke 'news_lda.csv'")
